```rust
use serde::{Deserialize, Serialize};
use std::path::PathBuf;
use tauri::AppHandle;
use xcap::{Monitor, Window};

#[derive(Debug, Serialize, Deserialize, Clone)]
pub struct WindowInfo {
    pub id: u32,
    pub title: String,
    pub app_name: String,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct CaptureRegionArgs {
    pub x: i32,
    pub y: i32,
    pub width: u32,
    pub height: u32,
}

fn write_to_temp(app: &AppHandle, image: xcap::image::RgbaImage, name: &str) -> Result<String, String> {
    let temp_dir = app.path().temp_dir().map_err(|e| e.to_string())?;
    let path: PathBuf = temp_dir.join(format!("screenshots_{}.png", name));
    image.save(&path).map_err(|e| e.to_string())?;
    Ok(path.to_string_lossy().to_string())
}

/// Capture the primary monitor. Returns absolute path to temp PNG.
#[tauri::command]
pub async fn capture_fullscreen(app: AppHandle) -> Result<String, String> {
    let monitors = Monitor::all().map_err(|e| format!("Failed to list monitors: {}", e))?;
    let primary = monitors.into_iter()
        .find(|m| m.is_primary().unwrap_or(false))
        .ok_or("No primary monitor found")?;
    let image = primary.capture_image().map_err(|e| format!("Capture failed: {}", e))?;
    let ts = std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH)
        .unwrap_or_default()
        .as_millis();
    write_to_temp(&app, image, &format!("fullscreen_{}", ts))
}

/// Capture a specific region. Coordinates in physical pixels (caller multiplies by devicePixelRatio).
/// Validates dimensions are non-zero before calling xcap.
#[tauri::command]
pub async fn capture_region(app: AppHandle, args: CaptureRegionArgs) -> Result<String, String> {
    if args.width == 0 || args.height == 0 {
        return Err("Region width and height must be greater than zero".to_string());
    }
    let monitors = Monitor::all().map_err(|e| format!("Failed to list monitors: {}", e))?;
    let primary = monitors.into_iter()
        .find(|m| m.is_primary().unwrap_or(false))
        .ok_or("No primary monitor found")?;
    let image = primary
        .capture_area(args.x, args.y, args.width, args.height)
        .map_err(|e| format!("Region capture failed: {}", e))?;
    let ts = std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH)
        .unwrap_or_default()
        .as_millis();
    write_to_temp(&app, image, &format!("region_{}", ts))
}

/// Return metadata for all visible, titled windows.
#[tauri::command]
pub async fn list_windows() -> Result<Vec<WindowInfo>, String> {
    let windows = Window::all().map_err(|e| format!("Failed to list windows: {}", e))?;
    let result = windows
        .into_iter()
        .filter(|w| !w.title().is_empty() && !w.is_minimized().unwrap_or(false))
        .map(|w| WindowInfo {
            id: w.id(),
            title: w.title().to_string(),
            app_name: w.app_name().to_string(),
        })
        .collect();
    Ok(result)
}

/// Capture a specific window by ID. Returns path to temp PNG.
#[tauri::command]
pub async fn capture_window(app: AppHandle, window_id: u32) -> Result<String, String> {
    let windows = Window::all().map_err(|e| format!("Failed to list windows: {}", e))?;
    let target = windows
        .into_iter()
        .find(|w| w.id() == window_id)
        .ok_or_else(|| format!("Window {} not found", window_id))?;
    let image = target.capture_image().map_err(|e| format!("Window capture failed: {}", e))?;
    let ts = std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH)
        .unwrap_or_default()
        .as_millis();
    write_to_temp(&app, image, &format!("window_{}", ts))
}

/// macOS only: check Screen Recording permission before every capture.
/// Returns true if granted. If false, the frontend shows recovery UI.
#[cfg(target_os = "macos")]
#[tauri::command]
pub async fn check_screen_permission(app: AppHandle) -> Result<bool, String> {
    use tauri_plugin_macos_permissions::PermissionsExt;
    let granted = app.check_screen_recording_permission();
    if !granted {
        app.request_screen_recording_permission();
    }
    Ok(granted)
}

// Non-macOS stub so the command is always registered in lib.rs
#[cfg(not(target_os = "macos"))]
#[tauri::command]
pub async fn check_screen_permission(_app: AppHandle) -> Result<bool, String> {
    Ok(true) // Permission not required on Windows/Linux
}
```

```rust
use tauri::{
    menu::{Menu, MenuItem, PredefinedMenuItem},
    tray::{MouseButton, MouseButtonState, TrayIconBuilder, TrayIconEvent},
    Manager,
};
use tauri_plugin_global_shortcut::{GlobalShortcutExt, Shortcut, ShortcutState};

mod commands;

#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
pub struct PlatformFlags {
    pub is_wayland: bool,
}

#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
pub struct HotkeyConfig {
    pub capture_region: String,
    pub capture_fullscreen: String,
    pub capture_window: String,
}

impl Default for HotkeyConfig {
    fn default() -> Self {
        Self {
            capture_region: "CommandOrControl+Shift+3".to_string(),
            capture_fullscreen: "CommandOrControl+Shift+4".to_string(),
            capture_window: "CommandOrControl+Shift+5".to_string(),
        }
    }
}

fn setup_tray(app: &tauri::AppHandle) -> tauri::Result<()> {
    let capture_region = MenuItem::with_id(app, "capture-region", "Capture Region", true, None::<&str>)?;
    let capture_screen = MenuItem::with_id(app, "capture-screen", "Capture Full Screen", true, None::<&str>)?;
    let capture_window_item = MenuItem::with_id(app, "capture-window", "Capture Window", true, None::<&str>)?;
    let sep = PredefinedMenuItem::separator(app)?;
    let settings = MenuItem::with_id(app, "settings", "Settings", true, None::<&str>)?;
    let quit = MenuItem::with_id(app, "quit", "Quit Screenshots", true, None::<&str>)?;

    let menu = Menu::with_items(app, &[
        &capture_region,
        &capture_screen,
        &capture_window_item,
        &sep,
        &settings,
        &quit,
    ])?;

    TrayIconBuilder::new()
        .icon(app.default_window_icon().unwrap().clone())
        .menu(&menu)
        .menu_on_left_click(false)
        .on_menu_event(|app, event| match event.id.as_ref() {
            "capture-region" => { let _ = app.emit("capture:region", ()); }
            "capture-screen" => { let _ = app.emit("capture:screen", ()); }
            "capture-window" => { let _ = app.emit("capture:window", ()); }
            "settings" => {
                if let Some(win) = app.get_webview_window("main") {
                    let _ = app.emit("navigate", "/settings");
                    let _ = win.show();
                    let _ = win.set_focus();
                }
            }
            "quit" => app.exit(0),
            _ => {}
        })
        .on_tray_icon_event(|tray, event| {
            if let TrayIconEvent::Click {
                button: MouseButton::Left,
                button_state: MouseButtonState::Up,
                ..
            } = event
            {
                let app = tray.app_handle();
                if let Some(win) = app.get_webview_window("main") {
                    let _ = win.show();
                    let _ = win.set_focus();
                }
            }
        })
        .build(app)?;
    Ok(())
}

fn register_shortcuts(app: &tauri::AppHandle, config: &HotkeyConfig) {
    // Wayland guard: if is_wayland is managed, skip registration
    if let Some(flags) = app.try_state::<PlatformFlags>() {
        if flags.is_wayland {
            return;
        }
    }

    let _ = app.global_shortcut().unregister_all();

    let region_str = config.capture_region.clone();
    let fullscreen_str = config.capture_fullscreen.clone();
    let window_str = config.capture_window.clone();

    let Ok(region_sc) = region_str.parse::<Shortcut>() else { return; };
    let Ok(fullscreen_sc) = fullscreen_str.parse::<Shortcut>() else { return; };
    let Ok(window_sc) = window_str.parse::<Shortcut>() else { return; };

    let handle = app.clone();
    let _ = app.global_shortcut().on_shortcuts(
        &[region_sc.clone(), fullscreen_sc.clone(), window_sc.clone()],
        move |_app, shortcut, event| {
            if event.state() != ShortcutState::Pressed {
                return;
            }
            if shortcut == &region_sc {
                let _ = handle.emit("capture:region", ());
            } else if shortcut == &fullscreen_sc {
                let _ = handle.emit("capture:screen", ());
            } else if shortcut == &window_sc {
                let _ = handle.emit("capture:window", ());
            }
        },
    );
}

/// Called from the frontend settings page to update hotkeys at runtime.
#[tauri::command]
pub fn update_hotkeys(app: tauri::AppHandle, config: HotkeyConfig) -> Result<(), String> {
    register_shortcuts(&app, &config);
    Ok(())
}

pub fn run() {
    let is_wayland = std::env::var("WAYLAND_DISPLAY").is_ok();

    tauri::Builder::default()
        .plugin(tauri_plugin_store::Builder::default().build())
        .plugin(tauri_plugin_window_state::Builder::default().build())
        .plugin(tauri_plugin_global_shortcut::Builder::new().build())
        .manage(PlatformFlags { is_wayland })
        .setup(|app| {
            let handle = app.handle();

            // System tray
            setup_tray(handle)?;

            // Register default hotkeys (skipped automatically on Wayland)
            register_shortcuts(handle, &HotkeyConfig::default());

            // Emit platform flags to frontend
            handle.emit("platform:flags", PlatformFlags { is_wayland })?;

            Ok(())
        })
        .invoke_handler(tauri::generate_handler![
            commands::capture::capture_fullscreen,
            commands::capture::capture_region,
            commands::capture::list_windows,
            commands::capture::capture_window,
            commands::capture::check_screen_permission,
            update_hotkeys,
        ])
        .run(tauri::generate_context!())
        .expect("error while running tauri application");
}
```

**src/hooks/useCaptureFlow.ts** — orchestrates the full capture lifecycle:

```typescript
import { useEffect, useCallback } from 'react';
import { listen } from '@tauri-apps/api/event';
import { convertFileSrc } from '@tauri-apps/api/core';
import { getCurrentWindow } from '@tauri-apps/api/window';
import {
  captureFullscreen,
  captureRegion,
  captureWindow,
  checkScreenPermission,
  type CaptureRegionArgs,
} from '../ipc/capture';
import { useAppStore } from '../stores/app.store';

export function useCaptureFlow() {
  const { setCaptureState, setLastCapturePath, isWayland } = useAppStore();

  const performFullscreen = useCallback(async () => {
    try {
      // macOS: always check permission before capture
      if ((window as any).__TAURI_OS_PLUGIN_INTERNALS__?.platform === 'macos') {
        const granted = await checkScreenPermission();
        if (!granted) return; // permission dialog shown to user by Rust
      }
      setCaptureState('capturing');
      const path = await captureFullscreen();
      setLastCapturePath(convertFileSrc(path));
      setCaptureState('captured');
      // Ensure main window is visible with the capture result
      const win = getCurrentWindow();
      await win.show();
      await win.setFocus();
    } catch (err) {
      console.error('Fullscreen capture failed:', err);
      setCaptureState('idle');
    }
  }, [setCaptureState, setLastCapturePath]);

  const initiateRegion = useCallback(async () => {
    setCaptureState('selecting-region');
    // Show the capture-overlay window — it handles its own completion
    const { WebviewWindow } = await import('@tauri-apps/api/webviewWindow');
    const overlay = WebviewWindow.getByLabel('capture-overlay');
    if (overlay) {
      await overlay.show();
      await overlay.setFocus();
    }
  }, [setCaptureState]);

  const initiateWindowPicker = useCallback(async () => {
    setCaptureState('selecting-window');
    // Emit event so WindowPicker component shows in the main window
    const { emit } = await import('@tauri-apps/api/event');
    await emit('ui:show-window-picker', null);
  }, [setCaptureState]);

  useEffect(() => {
    const listeners = [
      listen('capture:screen', () => performFullscreen()),
      listen('capture:region', () => initiateRegion()),
      listen('capture:window', () => initiateWindowPicker()),
    ];

    return () => {
      listeners.forEach((p) => p.then((fn) => fn()));
    };
  }, [performFullscreen, initiateRegion, initiateWindowPicker]);

  return { performFullscreen, initiateRegion, initiateWindowPicker };
}
```

**src/components/capture/RegionOverlay.tsx** — transparent fullscreen overlay for rubber-band region selection. This component is the root of the `capture-overlay` window:

```tsx
import { useRef, useState, useCallback, useEffect } from 'react';
import { invoke } from '@tauri-apps/api/core';
import { convertFileSrc } from '@tauri-apps/api/core';
import { getCurrentWindow } from '@tauri-apps/api/window';
import { emit } from '@tauri-apps/api/event';
import { captureRegion } from '../../ipc/capture';

interface SelectionRect {
  x: number;
  y: number;
  width: number;
  height: number;
}

export function RegionOverlay() {
  const [isDragging, setIsDragging] = useState(false);
  const [startPos, setStartPos] = useState<{ x: number; y: number } | null>(null);
  const [selectionRect, setSelectionRect] = useState<SelectionRect | null>(null);
  const containerRef = useRef<HTMLDivElement>(null);

  const handleMouseDown = useCallback((e: React.MouseEvent) => {
    e.preventDefault();
    setIsDragging(true);
    setStartPos({ x: e.clientX, y: e.clientY });
    setSelectionRect(null);
  }, []);

  const handleMouseMove = useCallback(
    (e: React.MouseEvent) => {
      if (!isDragging || !startPos) return;
      const x = Math.min(e.clientX, startPos.x);
      const y = Math.min(e.clientY, startPos.y);
      const width = Math.abs(e.clientX - startPos.x);
      const height = Math.abs(e.clientY - startPos.y);
      setSelectionRect({ x, y, width, height });
    },
    [isDragging, startPos]
  );

  const handleMouseUp = useCallback(async () => {
    if (!selectionRect || selectionRect.width < 4 || selectionRect.height < 4) {
      setIsDragging(false);
      setStartPos(null);
      setSelectionRect(null);
      return;
    }

    setIsDragging(false);
    const dpr = window.devicePixelRatio || 1;

    try {
      // Close overlay immediately so it is not in the capture
      await getCurrentWindow().hide();

      const path = await captureRegion({
        x: Math.round(selectionRect.x * dpr),
        y: Math.round(selectionRect.y * dpr),
        width: Math.round(selectionRect.width * dpr),
        height: Math.round(selectionRect.height * dpr),
      });

      // Notify main window of the result
      await emit('capture:complete', { path: convertFileSrc(path) });

      // Close the overlay window
      await getCurrentWindow().close();
    } catch (err) {
      console.error('Region capture failed:', err);
      await getCurrentWindow().hide();
    }
  }, [selectionRect]);

  // Escape key cancels selection
  useEffect(() => {
    const handleKey = async (e: KeyboardEvent) => {
      if (e.key === 'Escape') {
        await getCurrentWindow().hide();
      }
    };
    window.addEventListener('keydown', handleKey);
    return () => window.removeEventListener('keydown', handleKey);
  }, []);

  return (
    <div
      ref={containerRef}
      className="fixed inset-0 cursor-crosshair select-none"
      style={{ background: 'rgba(0,0,0,0.15)' }}
      onMouseDown={handleMouseDown}
      onMouseMove={handleMouseMove}
      onMouseUp={handleMouseUp}
    >
      {selectionRect && (
        <div
          className="absolute border-2 border-blue-400 bg-blue-400/10"
          style={{
            left: selectionRect.x,
            top: selectionRect.y,
            width: selectionRect.width,
            height: selectionRect.height,
          }}
        >
          <span className="absolute -top-6 left-0 text-white text-xs bg-blue-500 px-1 py-0.5 rounded">
            {Math.round(selectionRect.width)} × {Math.round(selectionRect.height)}
          </span>
        </div>
      )}
      <div className="absolute top-4 left-1/2 -translate-x-1/2 bg-black/60 text-white text-xs px-3 py-1.5 rounded-full">
        Drag to select a region • Press Esc to cancel
      </div>
    </div>
  );
}
```

```tsx
import { useState } from 'react';
import { useAppStore, type HotkeyConfig } from '../../stores/app.store';

const HOTKEY_LABELS: Record<keyof HotkeyConfig, string> = {

  captureFullscreen: 'Capture Full Screen',
  captureWindow: 'Capture Window',

export function SettingsPage() {

  const [saving, setSaving] = useState(false);
```

```tsx
    const unlisten = listen<{ path: string }>('capture:complete', (event) => {
```

```tsx
import { useEffect, useState } from 'react';
import { listen } from '@tauri-apps/api/event';
import { useAppStore } from './stores/app.store';
import { useCaptureFlow } from './hooks/useCaptureFlow';
import { WaylandBanner } from './components/shell/WaylandBanner';
import { TraySetup } from './components/shell/TraySetup';
import { SettingsPage } from './components/shell/SettingsPage';
import { WindowPicker } from './components/capture/WindowPicker';

interface PlatformFlags {
  is_wayland: boolean;
}

export default function App() {
  const setWayland = useAppStore((s) => s.setWayland);
  const setLastCapturePath = useAppStore((s) => s.setLastCapturePath);
  const setCaptureState = useAppStore((s) => s.setCaptureState);
  const lastCapturePath = useAppStore((s) => s.lastCapturePath);
  const [showSettings, setShowSettings] = useState(false);
  const [showWindowPicker, setShowWindowPicker] = useState(false);

  // Activate the capture event listeners
  useCaptureFlow();

  useEffect(() => {
    const unlisteners = [
      listen<PlatformFlags>('platform:flags', (e) => {
        setWayland(e.payload.is_wayland);
      }),
      listen('navigate', (e) => {
        if (e.payload === '/settings') setShowSettings(true);
      }),
      listen('ui:show-window-picker', () => {
        setShowWindowPicker(true);
      }),
    ];
    return () => { unlisteners.forEach((p) => p.then((fn) => fn())); };
  }, [setWayland]);

  return (
    <div className="flex flex-col h-screen bg-neutral-950 text-neutral-100 overflow-hidden">
      {/* Side-effect components */}
      <TraySetup />

      {/* Platform banner */}
      <WaylandBanner />

      {/* Main content */}
      <main className="flex-1 flex items-center justify-center overflow-hidden relative">
        {showSettings ? (
          <div className="w-full h-full overflow-y-auto">
            <div className="flex items-center gap-2 px-6 pt-6 pb-2">
              <button
                onClick={() => setShowSettings(false)}
                className="text-neutral-400 hover:text-neutral-100 text-xs"
              >
                ← Back
              </button>
            </div>
            <SettingsPage />
          </div>
        ) : lastCapturePath ? (
          <div className="p-6 flex flex-col items-center gap-4 max-w-full max-h-full">
            <img
              src={lastCapturePath}
              alt="Captured screenshot"
              className="max-w-full max-h-[calc(100vh-120px)] object-contain rounded-lg shadow-2xl"
            />
            <div className="flex gap-3">
              <button
                onClick={() => { setLastCapturePath(null); setCaptureState('idle'); }}
                className="px-4 py-2 bg-neutral-800 hover:bg-neutral-700 text-sm rounded-lg transition-colors"
              >
                New Capture
              </button>
            </div>
          </div>
        ) : (
          <div className="text-center">
            <p className="text-neutral-400 text-sm mb-2">Ready to capture</p>
            <p className="text-neutral-600 text-xs">Use the system tray or keyboard shortcuts to take a screenshot</p>
          </div>
        )}
      </main>

      {/* Window picker modal */}
      {showWindowPicker && (
        <WindowPicker
          onCapture={(src) => {
            setLastCapturePath(src);
            setCaptureState('captured');
          }}
          onClose={() => {
            setShowWindowPicker(false);
            setCaptureState('idle');
          }}
        />
      )}
    </div>
  );
}
```
