export type ToolMode =
  | 'select' | 'arrow' | 'rect' | 'ellipse' | 'text' | 'emoji'
  | 'blur-region' | 'pixelate-region' | 'crop';

  id: string;
  src: string;          // file:// or object URL — serializable string only
  x: number;
  y: number;
  width: number;
  height: number;
  rotation: number;
  scaleX: number;       // use -1 to flip horizontally
  scaleY: number;       // use -1 to flip vertically
  cornerRadius: number;
  shadowBlur: number;
  shadowColor: string;
  shadowOffsetX: number;
  shadowOffsetY: number;
  shadowOpacity: number;
  strokeColor: string;  // inset border color — empty string = no border
  strokeWidth: number;
  opacity: number;
  crop?: { x: number; y: number; width: number; height: number }; // source pixel coords
}

export type AnnotationShape =
  | { kind: 'arrow';   id: string; points: number[]; tension: number; stroke: string; strokeWidth: number; pointerLength: number; pointerWidth: number }
  | { kind: 'rect';    id: string; x: number; y: number; width: number; height: number; fill: string; stroke: string; strokeWidth: number; cornerRadius: number; rotation: number }
  | { kind: 'ellipse'; id: string; x: number; y: number; radiusX: number; radiusY: number; fill: string; stroke: string; strokeWidth: number; rotation: number }
  | { kind: 'text';    id: string; x: number; y: number; text: string; fontSize: number; fontFamily: string; fill: string; stroke: string; strokeWidth: number; rotation: number }
  | { kind: 'emoji';   id: string; x: number; y: number; emoji: string; fontSize: number; rotation: number };

export interface CanvasBackground {
  type: 'solid' | 'linear-gradient' | 'radial-gradient' | 'image';
  color?: string;
  gradientStops?: Array<{ offset: number; color: string }>;
  gradientAngle?: number;
  imageSrc?: string;
  blurRadius?: number;
  noiseAmount?: number; // 0–1
}

  canvasWidth: number;
  canvasHeight: number;

  annotations: AnnotationShape[];

  background: CanvasBackground;

  addImage(img: CanvasImage): void;
  updateImage(id: string, patch: Partial<CanvasImage>): void;
  removeImage(id: string): void;
  addAnnotation(ann: AnnotationShape): void;
  updateAnnotation(id: string, patch: Partial<AnnotationShape>): void;
  removeAnnotation(id: string): void;
  setSelected(id: string | null): void;
  setBackground(bg: Partial<CanvasBackground>): void;

export const useToolStore: UseBoundStore<...>;

**src/components/canvas/BackgroundLayer.tsx** — Renders the background based on `useCanvasStore(s => s.background)`:
- `solid`: `<Rect fill={color} width={w} height={h} listening={false} />`
- `linear-gradient`: `<Rect>` with `fillLinearGradientStartPoint`, `fillLinearGradientEndPoint` (computed from `gradientAngle` using the `angleToPoints` helper), `fillLinearGradientColorStops` (flat array: `[offset0, color0, offset1, color1, ...]`)
- `radial-gradient`: `<Rect>` with `fillRadialGradientStartPoint={center}`, `fillRadialGradientEndRadius={Math.max(w,h)/2}`
- `image`: `<KonvaImage>` using `useImage(imageSrc, 'anonymous')`. When `blurRadius > 0`: call `bgRef.current.cache()` then `.filters([Konva.Filters.Blur])` and `.blurRadius(value)` in a `useEffect` that re-runs when `blurRadius` or `bgImage` changes. For noise (BG-06): overlay a second `<Rect ref={noiseRef}>` with `fill="transparent"` that has `Konva.Filters.Noise` applied with `noise(noiseAmount)` — only cache/apply when `noiseAmount > 0`
- The entire layer must be `listening={false}` — background never receives pointer events
- Wrap the single gradient/solid Rect in a ref and call `cache()` + noise filter in the same useEffect when `noiseAmount` changes (for non-image backgrounds)

**src/components/canvas/nodes/ScreenshotNode.tsx** — Single image component receiving `img: CanvasImage` and `isSelected: boolean`:
- Load image via `useImage(img.src)` hook from `use-image` package
- Render `<KonvaImage>` with ALL properties from `CanvasImage` mapped directly to Konva props: `cornerRadius`, `shadowBlur`, `shadowColor`, `shadowOffsetX`, `shadowOffsetY`, `shadowOpacity`, `scaleX`, `scaleY`, `crop`
- Render a sibling `<Rect>` for the inset border (STYL-05) — same position/size/cornerRadius as the image, `fill="transparent"`, `stroke={img.strokeColor}`, `strokeWidth={img.strokeWidth}`, `listening={false}`. Only render when `img.strokeWidth > 0`
- When selected, render a `<Transformer ref={trRef}>` attached to the image node. In `onTransformEnd`: CRITICAL — Transformer changes scale not dimensions. Read back `node.width() * node.scaleX()` for new width, `node.height() * node.scaleY()` for new height, `node.rotation()`, `node.x()`, `node.y()`. Call `updateImage(img.id, { x, y, width, height, rotation, scaleX: 1, scaleY: 1 })`. Then reset the node: `node.scaleX(1); node.scaleY(1)`. This normalization is mandatory — see RESEARCH.md Pitfall 2
- `draggable` prop: only true when `isSelected && activeTool === 'select'`
- On drag end: call `updateImage(img.id, { x: node.x(), y: node.y() })`
- For flipping (EDIT-02, EDIT-03): scaleX=-1 mirrors horizontally. The image x/y offset must be adjusted when scale is negative — set `offsetX = width` when `scaleX < 0`, `offsetY = height` when `scaleY < 0`

**src/components/canvas/ScreenshotLayer.tsx** — Layer containing all ScreenshotNode instances:

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| File system → Canvas | User-supplied image files are opened via Tauri dialog and converted to webview URLs via `convertFileSrc()`. The path is trusted OS-provided; no injection surface here. |
| Drag-and-drop → Canvas stage | `dataTransfer.files` from browser DnD is browser-provided and filtered to image/* MIME types before processing. |
| Canvas state → Undo history | zundo snapshots the canvas store. Non-serializable objects crashing structured clone could corrupt undo history. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Tampering | CanvasStage DnD handler | mitigate | Filter `dataTransfer.files` to `image/*` MIME types only before calling `convertFileSrc()`. Reject non-image files with a user-visible toast. |
| T-02-02 | Denial of Service | BackgroundLayer noise filter | mitigate | Clamp `noiseAmount` to [0, 1] and `blurRadius` to [0, 40] in the store setter before passing to Konva — prevents runaway filter computation on malformed values. |
| T-02-03 | Tampering | zundo history store | mitigate | `partialize` in zundo config explicitly excludes action functions and non-serializable references. Add a TypeScript type assertion `satisfies SerializableCanvasState` to catch accidental non-serializable additions at compile time. |
| T-02-04 | Information Disclosure | colorAnalysis.ts (colorthief) | accept | Color extraction operates on already-displayed images — no new information disclosure beyond what the user already sees. CORS failure returns a safe fallback color. |
| T-02-05 | Tampering | openImageFile IPC wrapper | mitigate | Tauri dialog plugin returns only paths the user explicitly selected via OS native dialog. No path traversal possible — the user chose the file. Validate MIME by file extension as a secondary check before loading. |
</threat_model>
