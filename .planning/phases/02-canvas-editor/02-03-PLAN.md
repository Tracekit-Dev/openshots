Then implement `useHotkeys()` — a hook that mounts all tinykeys bindings:

```typescript
export function useHotkeys(onOpenShortcutsModal: () => void) {
  useEffect(() => {
    // Guard: prevent shortcuts firing while user types in an input or textarea
    function isTypingInInput(): boolean {
      const el = document.activeElement;
      if (!el) return false;
      const tag = el.tagName.toLowerCase();
      return tag === 'input' || tag === 'textarea' || (el as HTMLElement).isContentEditable;
    }

    function guard(handler: () => void) {
      return (e: KeyboardEvent) => {
        if (isTypingInInput()) return;
        e.preventDefault();
        handler();
      };
    }

    function deleteSelected() {
      const { selectedId, removeAnnotation, removeImage, setSelected, annotations, images } = useCanvasStore.getState();
      if (!selectedId) return;
      const isAnnotation = annotations.some((a) => a.id === selectedId);
      const isImage = images.some((img) => img.id === selectedId);
      if (isAnnotation) removeAnnotation(selectedId);
      else if (isImage) removeImage(selectedId);
      setSelected(null);
    }

    const unsub = tinykeys(window, {
      '$mod+z': guard(() => {
        useCanvasStore.temporal.getState().undo();
      }),
      '$mod+Shift+z': guard(() => {
        useCanvasStore.temporal.getState().redo();
      }),
      '$mod+y': guard(() => {
        useCanvasStore.temporal.getState().redo();
      }),
      'Delete':    guard(deleteSelected),
      'Backspace': guard(deleteSelected),
      'Escape':    guard(() => useToolStore.getState().setTool('select')),
      'v':         guard(() => useToolStore.getState().setTool('select')),
      'a':         guard(() => useToolStore.getState().setTool('arrow')),
      'r':         guard(() => useToolStore.getState().setTool('rect')),
      'e':         guard(() => useToolStore.getState().setTool('ellipse')),
      't':         guard(() => useToolStore.getState().setTool('text')),
      '?':         guard(onOpenShortcutsModal),
      '$mod+/':    guard(onOpenShortcutsModal),
    });

    return unsub; // tinykeys returns an unsubscribe/cleanup function
  }, [onOpenShortcutsModal]);
}
```

**src/components/toolbar/ShortcutsModal.tsx** — Shortcut reference dialog (KEYS-02):

Build using shadcn/ui `Dialog` component (already in the project via Plan 01's shadcn setup). The modal displays all entries from `SHORTCUT_REGISTRY` grouped by `category`.

Implementation:

```typescript
import { Dialog, DialogContent, DialogHeader, DialogTitle } from '@/components/ui/dialog';
import { SHORTCUT_REGISTRY, type ShortcutEntry } from '../../hooks/useHotkeys';

// Detect platform for display (Cmd vs Ctrl labels)
const isMac = navigator.platform.startsWith('Mac') || navigator.userAgent.includes('Mac');

interface ShortcutsModalProps {
  open: boolean;
  onClose: () => void;
}

export function ShortcutsModal({ open, onClose }: ShortcutsModalProps) {
  // Group entries by category
  const grouped = SHORTCUT_REGISTRY.reduce<Record<string, ShortcutEntry[]>>((acc, entry) => {
    if (!acc[entry.category]) acc[entry.category] = [];
    acc[entry.category].push(entry);
    return acc;
  }, {});

  const categoryOrder = ['History', 'Tools', 'Selection', 'View'];

  return (
    <Dialog open={open} onOpenChange={(o) => { if (!o) onClose(); }}>
      <DialogContent className="max-w-md">
        <DialogHeader>
          <DialogTitle>Keyboard Shortcuts</DialogTitle>
        </DialogHeader>

        <div className="space-y-4 py-2">
          {categoryOrder.map((category) => {
            const entries = grouped[category];
            if (!entries?.length) return null;
            return (
              <div key={category}>
                <h3 className="text-xs font-semibold uppercase tracking-wider text-muted-foreground mb-2">
                  {category}
                </h3>
                <div className="space-y-1">
                  {entries.map((entry, i) => (
                    <div key={i} className="flex items-center justify-between py-0.5">
                      <span className="text-sm text-foreground">{entry.description}</span>
                      <kbd className="inline-flex items-center rounded border border-border bg-muted px-1.5 py-0.5 text-xs font-mono text-muted-foreground">
                        {isMac ? entry.keys : (entry.keysWin ?? entry.keys)}
                      </kbd>
                    </div>
                  ))}
                </div>
              </div>
            );
          })}
        </div>
      </DialogContent>
    </Dialog>
  );
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Keyboard events → app actions | tinykeys processes raw keyboard events from window. Malicious key sequences could trigger undo floods or rapid tool switching but cannot exfiltrate data. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-11 | Denial of Service | useHotkeys — undo flood | accept | A user holding Cmd+Z can exhaust the 100-step undo limit (set via zundo's `limit` option in canvas.store.ts). This is expected behavior for a desktop app. The undo stack limit prevents unbounded memory growth. |
| T-02-12 | Tampering | useHotkeys — input guard bypass | mitigate | The `isTypingInInput()` guard checks `document.activeElement`. If a custom component captures focus without being an input/textarea, shortcuts could fire unexpectedly. Any future rich-text components or custom contenteditable elements must set `contentEditable="true"` so the guard catches them. Document this in CLAUDE.md if contenteditable is introduced later. |
| T-02-13 | Information Disclosure | ShortcutsModal | accept | The modal displays hardcoded shortcut descriptions. No user data or sensitive information is exposed. |
</threat_model>
