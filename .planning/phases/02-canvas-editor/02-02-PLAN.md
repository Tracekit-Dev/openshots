**src/workers/backgroundRemoval.worker.ts** — Web Worker that runs the RMBG-1.4 model:

```typescript
import { pipeline, RawImage } from '@huggingface/transformers';
import type { ImageSegmentationPipeline } from '@huggingface/transformers';

let segmenter: ImageSegmentationPipeline | null = null;

interface WorkerRequest {
  imageSrc: string;  // object URL or file:// URL for the image to process
}

interface WorkerResponse {
  status: 'loading' | 'processing' | 'done' | 'error';
  progress?: number;  // 0–100 during model loading
  resultUrl?: string; // object URL of the result PNG with alpha, when status='done'
  error?: string;     // when status='error'
}

self.onmessage = async (e: MessageEvent<WorkerRequest>) => {
  try {
    if (!segmenter) {
      self.postMessage({ status: 'loading', progress: 0 } satisfies WorkerResponse);
      segmenter = await pipeline(
        'image-segmentation',
        'briaai/RMBG-1.4',
        {
          device: 'webgpu',  // falls back to 'wasm' automatically if WebGPU unavailable
          progress_callback: (info: { progress?: number }) => {
            self.postMessage({ status: 'loading', progress: info.progress ?? 0 } satisfies WorkerResponse);
          },
        }
      ) as ImageSegmentationPipeline;
    }

    self.postMessage({ status: 'processing' } satisfies WorkerResponse);

    const result = await segmenter(e.data.imageSrc);
    // result is an array; result[0].mask is the segmentation mask
    // The mask is a RawImage with width/height matching the input

    // Apply mask as alpha channel to original image
    const originalImage = await RawImage.fromURL(e.data.imageSrc);
    const mask = result[0].mask;  // RawImage, single-channel

    // Create RGBA output
    const outputData = new Uint8ClampedArray(originalImage.width * originalImage.height * 4);
    for (let i = 0; i < originalImage.width * originalImage.height; i++) {
      outputData[i * 4 + 0] = originalImage.data[i * 3 + 0]; // R
      outputData[i * 4 + 1] = originalImage.data[i * 3 + 1]; // G
      outputData[i * 4 + 2] = originalImage.data[i * 3 + 2]; // B
      outputData[i * 4 + 3] = mask.data[i];                  // Alpha from mask
    }

    // Draw to OffscreenCanvas → Blob → Object URL
    const canvas = new OffscreenCanvas(originalImage.width, originalImage.height);
    const ctx = canvas.getContext('2d')!;
    const imageData = new ImageData(outputData, originalImage.width, originalImage.height);
    ctx.putImageData(imageData, 0, 0);
    const blob = await canvas.convertToBlob({ type: 'image/png' });
    const resultUrl = URL.createObjectURL(blob);

    self.postMessage({ status: 'done', resultUrl } satisfies WorkerResponse);
  } catch (err) {
    self.postMessage({ status: 'error', error: String(err) } satisfies WorkerResponse);
  }
};
```

Note: The exact shape of the Transformers.js pipeline output for 'image-segmentation' must be verified at runtime. The `result[0].mask` API is documented for RMBG-1.4 in the @huggingface/transformers v4 docs. If the API surface differs, adapt using `Object.keys(result[0])` logged to console during development.

**src/hooks/useBackgroundRemoval.ts** — Hook that manages the Worker lifecycle and exposes the remove function:

```typescript
import { useRef, useEffect, useCallback, useState } from 'react';

type RemovalStatus = 'idle' | 'loading-model' | 'processing' | 'error';

export function useBackgroundRemoval() {
  const workerRef = useRef<Worker | null>(null);
  const [status, setStatus] = useState<RemovalStatus>('idle');
  const [progress, setProgress] = useState(0);

  useEffect(() => {
    workerRef.current = new Worker(
      new URL('../workers/backgroundRemoval.worker.ts', import.meta.url),
      { type: 'module' }
    );
    return () => {
      workerRef.current?.terminate();
      workerRef.current = null;
    };
  }, []);

  const removeBackground = useCallback(async (imageSrc: string): Promise<string> => {
    return new Promise((resolve, reject) => {
      if (!workerRef.current) {
        reject(new Error('Worker not initialized'));
        return;
      }

      workerRef.current.onmessage = (e) => {
        const msg = e.data;
        if (msg.status === 'loading') {
          setStatus('loading-model');
          setProgress(msg.progress ?? 0);
        } else if (msg.status === 'processing') {
          setStatus('processing');
        } else if (msg.status === 'done') {
          setStatus('idle');
          resolve(msg.resultUrl);
        } else if (msg.status === 'error') {
          setStatus('error');
          reject(new Error(msg.error));
        }
      };

      workerRef.current.postMessage({ imageSrc });
    });
  }, []);

  return { removeBackground, status, progress };
}
```
