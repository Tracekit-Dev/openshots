**Create src/components/canvas/GuidesLayer.tsx:**

New file. Renders red alignment guide lines based on a `guides` prop.

```typescript
import { Layer, Line } from "react-konva";

export interface Guide {
  lineGuide: number;    // position of the line (x or y coordinate)
  orientation: "V" | "H"; // vertical or horizontal
}

interface GuidesLayerProps {
  guides: Guide[];
  canvasWidth: number;
  canvasHeight: number;
}

export default function GuidesLayer({ guides, canvasWidth, canvasHeight }: GuidesLayerProps) {
  return (
    <Layer listening={false}>
      {guides.map((guide, i) => {
        if (guide.orientation === "V") {
          return (
            <Line
              key={`v-${i}`}
              points={[guide.lineGuide, 0, guide.lineGuide, canvasHeight]}
              stroke="#ef4444"
              strokeWidth={1}
              listening={false}
            />
          );
        }
        return (
          <Line
            key={`h-${i}`}
            points={[0, guide.lineGuide, canvasWidth, guide.lineGuide]}
            stroke="#ef4444"
            strokeWidth={1}
            listening={false}
          />
        );
      })}
    </Layer>
  );
}
```

3. Add the snap detection helper functions inside the component (or above it):

   ```typescript
   const SNAP_THRESHOLD = 8; // per D-10

   function getLineGuideStops(
     allImages: CanvasImage[],
     skipId: string,
     canvasWidth: number,
     canvasHeight: number,
   ): { v: number[]; h: number[] } {
     const v: number[] = [0, canvasWidth / 2, canvasWidth];
     const h: number[] = [0, canvasHeight / 2, canvasHeight];
     for (const img of allImages) {
       if (img.id === skipId) continue;
       // Use stored positions. offsetX/offsetY = width/2, height/2, so:
       const left = img.x - img.width / 2;
       const right = img.x + img.width / 2;
       const top = img.y - img.height / 2;
       const bottom = img.y + img.height / 2;
       v.push(left, img.x, right);
       h.push(top, img.y, bottom);
     }
     return { v, h };
   }
   ```

4. Add onDragMove handler:
   ```typescript
   const handleDragMove = (e: Konva.KonvaEventObject<DragEvent>) => {
     const node = e.target;
     const box = node.getClientRect({ relativeTo: node.getParent()?.getParent() });
     // box gives us the bounding rect in layer coordinates
     const absPos = node.absolutePosition();
     const stage = node.getStage();
     if (!stage) return;

     // Get the node's edges in canvas space
     const nodeLeft = node.x() - totalW / 2;
     const nodeRight = node.x() + totalW / 2;
     const nodeTop = node.y() - totalH / 2;
     const nodeBottom = node.y() + totalH / 2;
     const nodeCenterX = node.x();
     const nodeCenterY = node.y();

     const stops = getLineGuideStops(allImages, data.id, canvasWidth, canvasHeight);
     const newGuides: Guide[] = [];
     let snapX: number | null = null;
     let snapY: number | null = null;

     // Check vertical (x-axis) snaps
     const nodeVEdges = [
       { pos: nodeLeft, label: "left" },
       { pos: nodeCenterX, label: "center" },
       { pos: nodeRight, label: "right" },
     ];
     for (const edge of nodeVEdges) {
       for (const stop of stops.v) {
         if (Math.abs(edge.pos - stop) < SNAP_THRESHOLD) {
           snapX = stop - edge.pos + node.x();
           newGuides.push({ lineGuide: stop, orientation: "V" });
           break;
         }
       }
       if (snapX !== null) break;
     }

     // Check horizontal (y-axis) snaps
     const nodeHEdges = [
       { pos: nodeTop, label: "top" },
       { pos: nodeCenterY, label: "center" },
       { pos: nodeBottom, label: "bottom" },
     ];
     for (const edge of nodeHEdges) {
       for (const stop of stops.h) {
         if (Math.abs(edge.pos - stop) < SNAP_THRESHOLD) {
           snapY = stop - edge.pos + node.y();
           newGuides.push({ lineGuide: stop, orientation: "H" });
           break;
         }
       }
       if (snapY !== null) break;
     }

     // Apply snap position
     if (snapX !== null || snapY !== null) {
       node.position({
         x: snapX ?? node.x(),
         y: snapY ?? node.y(),
       });
     }

     setGuides(newGuides);
   };
   ```
