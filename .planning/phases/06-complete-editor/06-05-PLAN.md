   ```rust
   fn pixelate_region(
       img: &mut ImageBuffer<Rgba<u8>, Vec<u8>>,
       rx: u32, ry: u32, rw: u32, rh: u32,
       block_size: u32,
   ) {
       let block = block_size.max(1);
       let img_w = img.width();
       let img_h = img.height();
       // Clamp region to image bounds
       let rx = rx.min(img_w.saturating_sub(1));
       let ry = ry.min(img_h.saturating_sub(1));
       let rw = rw.min(img_w.saturating_sub(rx));
       let rh = rh.min(img_h.saturating_sub(ry));

       let mut by = 0;
       while by < rh {
           let mut bx = 0;
           while bx < rw {
               let bw = block.min(rw - bx);
               let bh = block.min(rh - by);
               let (mut sr, mut sg, mut sb, mut sa) = (0u64, 0u64, 0u64, 0u64);
               let mut count = 0u64;
               for dy in 0..bh {
                   for dx in 0..bw {
                       let px = img.get_pixel(rx + bx + dx, ry + by + dy);
                       sr += px[0] as u64;
                       sg += px[1] as u64;
                       sb += px[2] as u64;
                       sa += px[3] as u64;
                       count += 1;
                   }
               }
               if count > 0 {
                   let avg = Rgba([
                       (sr / count) as u8,
                       (sg / count) as u8,
                       (sb / count) as u8,
                       (sa / count) as u8,
                   ]);
                   for dy in 0..bh {
                       for dx in 0..bw {
                           img.put_pixel(rx + bx + dx, ry + by + dy, avg);
                       }
                   }
               }
               bx += block;
           }
           by += block;
       }
   }
   ```

5. After creating `final_img` (after the resize block around line 30, before the encoding block), add privacy region processing:
   ```rust
   // Process privacy regions per D-14, D-16, D-17
   if let Some(regions) = &privacy_regions {
       for region in regions {
           // Coords are already in pixel-space (frontend scales them)
           let rx = region.x.min(final_w.saturating_sub(1));
           let ry = region.y.min(final_h.saturating_sub(1));
           let rw = region.width.min(final_w.saturating_sub(rx));
           let rh = region.height.min(final_h.saturating_sub(ry));

           if rw == 0 || rh == 0 { continue; }

           match region.region_type.as_str() {
               "blur" => {
                   // Per D-16: sigma = intensity / 2.0 (intensity 1-40 -> sigma 0.5-20.0)
                   let sigma = region.intensity as f32 / 2.0;
                   // Extract sub-image, blur it, paste back
                   let sub = image::imageops::crop_imm(&final_img, rx, ry, rw, rh).to_image();
                   let blurred = gaussian_blur_f32(&sub, sigma);
                   image::imageops::replace(&mut final_img, &blurred, rx as i64, ry as i64);
               }
               "pixelate" => {
                   // Per D-17: block size = 8 * intensity
                   pixelate_region(&mut final_img, rx, ry, rw, rh, 8 * region.intensity);
               }
               _ => {} // Unknown type, skip
           }
       }
   }
   ```
